Type of Abstract Syntax Trees


type bop =
| Plus
| Minus
| Times
| Div 
type cmp =
| Strict
| Loose 
type base_type =
| IntT
| RealT
| BoolT 
val pp_base_type : Format.formatter -> base_type -> unit
type uexpr =
| UCst of Location.t * (Q.t * base_type option) (*
n
*)
| UVar of Location.t * Name.t (*
v
*)
| UBinop of Location.t * bop * uexpr * uexpr (*
expr + expr,...
*)
| URand of Location.t * base_type * Q.t * Q.t (*
rand(n, n)
*) 
type uguard = uexpr * cmp 
type ustm =
| UAsn of Location.t * Name.t * uexpr
| UAsrt of Location.t * uguard
| USeq of Location.t * ustm * ustm
| UIte of Location.t * uguard * ustm * ustm
| UWhile of Location.t * uguard * ustm 
type expr_desc =
| Cst of Q.t (*
n
*)
| Var of Name.t (*
v
*)
| Binop of bop * expr * expr (*
expr + expr,...
*)
| Rand of Q.t * Q.t (*
rand(n, n)
*) 
type expr = {
   expr_desc : expr_desc;
   expr_loc : Location.t;
   expr_type : base_type;
} 
type guard = expr * cmp
expr >= 0 (or >)
type stm =
| Asn of Location.t * Name.t * expr (*
v = expr;
*)
| Asrt of Location.t * guard (*
assert(guard);
*)
| Seq of Location.t * stm * stm (*
stm stm
*)
| Ite of Location.t * guard * stm * stm (*
if (guard) { stm } else { stm }
*)
| While of Location.t * guard * stm (*
while (guard) { stm }
*)

Various Utility Functions


val loc_of_expr : expr -> Location.t
loc_of_expr e returns location contained in expression e.
val loc_of_guard : guard -> Location.t
loc_of_guard g returns location contained in guard g.
val loc_of_stm : stm -> Location.t
loc_of_stm s returns location contained in statement s.
val vars_of_expr : expr -> Name.Set.t
vars_of_expr e returns the set of variables appearing in expression e.
val neg_guard : expr -> cmp -> expr * cmp
neg_guard e sl returns an expression e' such that guard e' >= 0 is equivalent to e > 0 is sl = Strict (resp >= if Loose). It depends on e type.
val vars_of_stm : stm -> Name.Set.t
vars_of_stm s returns the set of variables appearing in statement s.
val mk_expr : Location.t -> base_type -> expr_desc -> expr
val mk_cst_expr : Location.t -> base_type -> Q.t -> expr
