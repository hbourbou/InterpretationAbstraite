Module Utils

module Utils: sig .. end
Various utility functions.
val profile : (unit -> 'a) -> 'a * float
profile f executes the function f and returns both its result and the execution time in second.
val lines_of_string : string -> int
Returns the number of '\n' appearing in a string.
val output_filename_string : string option -> string
Returns string in option or "stdout" if None given.
val with_out_ch : string option -> (Pervasives.out_channel -> 'b) -> 'b
Open an output channel on given filename, gives it to the given function and eventually closes the channel. stdout is used if no filename is given.
val with_in_ch : string option -> (Pervasives.in_channel -> 'b) -> 'b
Same as with_out_ch for input channels (with stdin instead of stdout).
val select_param : string -> string -> string
select_param n p cuts the string p into n' and p' such that p = n':p' and n' doesn't contain the character ':'. Then, if n = n', p' is returned, otherwise the empty string is returned. If p doesn't contain the character ':', it is returned unchanged.
val warn_unknown_param : string -> string -> unit
warn_unknown_param n p does nothin if p is the empty string and outputs a warning otherwise (using Report.warning), saying that parameter p is unrecognized by domain n.
